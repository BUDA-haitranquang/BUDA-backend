- Thống kê doanh thu theo khách hàng (TOP x):
SELECT Name, sum(FinalCost) FROM Customers, Orders
WHERE Customers.CustomerID=Orders.CustomerID
AND CreationDate BETWEEN a AND b
AND UserID=y
GROUP BY CustomerID DESC LIMIT x;

- Thống kê giá trị trung bình của hóa đơn của tất cả khách hàng:
SELECT AVG(FinalCost) FROM Orders
WHERE CreationDate BETWEEN a AND b
AND UserID=y;
